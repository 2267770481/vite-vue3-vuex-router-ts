# 仪表盘
## 特点
```
仪表盘可以更直观的表现出某个指标的进度或实际情况
```
## 基本配置
```
series: [
  {
    name: '仪表盘',
    type: 'gauge',  
    data: [
      {
        value: 80,
        name: '当前值'
      }
    ],
    progress: {  // 进度展示
      show: true,
      color: 'auto'
    },
    min: 50,  // 仪表盘的刻度范围
    max: 150, // 仪表盘的刻度范围
  }
]
```
## 完整配置项
```
import { commonOption } from "./common";

export const gaugeOption ={
  title: commonOption.title('仪表盘'),
  toolbox: commonOption.toolbox,
  series: [
    {
      name: '仪表盘',
      type: 'gauge',  
      data: [
        {
          value: 80,
          name: '当前值'
        }
      ],
      progress: {
        show: true,
        color: 'auto'
      },
      min: 50,
      max: 150,
    }
  ]
}
```
